create table public.golden_set_library (
  id uuid not null default gen_random_uuid (),
  name text not null,
  category text null,
  description text null,
  properties jsonb null,
  status text null default 'draft'::text,
  enrichment_source text null,
  reviewer_note text null,
  created_at timestamp with time zone null default now(),
  outcome_type text null,
  failure_reason text null,
  ip_status text null default 'Unknown'::text,
  smiles_code text null,
  patent_id text null,
  patent_expiry date null,
  raw_data jsonb null default '{}'::jsonb,
  is_ai_extracted boolean null default false,
  ai_refined boolean null default false,
  processing_error text null,
  constraint golden_set_library_pkey primary key (id),
  constraint golden_set_library_outcome_type_check check (
    (
      outcome_type = any (
        array[
          'Success'::text,
          'Failure'::text,
          'Terminated'::text
        ]
      )
    )
  ),
  constraint golden_set_library_status_check check (
    (
      status = any (
        array['draft'::text, 'approved'::text, 'rejected'::text]
      )
    )
  )
) TABLESPACE pg_default;
